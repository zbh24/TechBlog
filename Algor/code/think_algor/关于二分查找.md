int binserchxia(vector<int>& nums,int target) {
    int low,high,mid;

    low = 0;
    high = nums.size() - 1;

    while (low <= high) {
      mid = low + (high -low) / 2;

      if (nums[mid] < target)
 	    low = mid + 1;
      else
  	     high = mid - 1;
    }
    if (nums[low] == target)

  }
